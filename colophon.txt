TWEETCARTS I WOULD LOVE TO RUN:

"infinite tunnel"

c={0,1,2,8,14,15,7}
fillp(0xa5a5)
function _draw()
 for w=3,68,.1 do
  a=4/w+t()/4
  k=145/w
  x=64+cos(a)*k
  y=64+sin(a)*k
  i=35/w+2+t()*3
  rect(x-w,y-w,x+w,y+w,f(i)*16+f(i+.5))
 end
end
function f(i)
 return c[flr(1.5+abs(6-i%12))]
end
